## Usage Examples

### Working with Verses

```ts
import type { Verse, VerseKey } from "@quranjs/api";

// Type-safe verse handling
const verse: Verse = await client.verses.findByKey("2:255" as VerseKey);

// Access optional properties safely
if (verse.translations) {
  verse.translations.forEach(translation => {
    console.log(translation.text);
  });
}

// Word-by-word access
if (verse.words) {
  verse.words.forEach(word => {
    console.log(`${word.textUthmani} - ${word.translation.text}`);
  });
}
```

### Working with Chapters

```ts
import type { Chapter } from "@quranjs/api";

const chapter: Chapter = await client.chapters.findById("1");

console.log(`${chapter.nameArabic} (${chapter.nameSimple})`);
console.log(`Verses: ${chapter.versesCount}`);
console.log(`Revelation place: ${chapter.revelationPlace}`);
```

### Translation Handling

```ts
import type { Translation } from "@quranjs/api";

// Filter translations by language
const englishTranslations = verse.translations?.filter(
  (t: Translation) => t.languageName === "English"
);

// Access translation metadata
englishTranslations?.forEach(translation => {
  console.log(`${translation.resourceName}: ${translation.text}`);
});
```
