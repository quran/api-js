## Usage Examples

### Working with Audio Responses

```ts
import type { AudioResponse, Segment } from "@quranjs/api";

// Get verse audio
const verse = await client.verses.findByKey("2:255", {
  audio: true,
  reciter: 7 // Mishary Alafasy
});

if (verse.audio) {
  const audio: AudioResponse = verse.audio;

  console.log(`Audio URL: ${audio.url}`);
  console.log(`Duration: ${audio.duration} seconds`);
  console.log(`Format: ${audio.format}`);

  // Word-level timing
  if (audio.segments) {
    audio.segments.forEach((segment: Segment, index: number) => {
      console.log(`Word ${index + 1}: ${segment.startTime}s - ${segment.endTime}s`);
    });
  }
}
```

### Working with Audio Segments

```ts
import type { Segment } from "@quranjs/api";

// Create audio player with word highlighting
function createAudioPlayer(segments: Segment[]) {
  return {
    highlightWord: (currentTime: number) => {
      const currentSegment = segments.find(
        segment => currentTime >= segment.startTime && currentTime <= segment.endTime
      );

      if (currentSegment) {
        // Highlight the current word
        return segments.indexOf(currentSegment);
      }

      return -1;
    },

    seekToWord: (wordIndex: number) => {
      if (segments[wordIndex]) {
        return segments[wordIndex].startTime;
      }
      return 0;
    }
  };
}
```

### Reciter Selection

```ts
import type { Reciter } from "@quranjs/api";

// Get available reciters
const reciters: Reciter[] = await client.audio.getReciters();

// Filter by style
const murattalReciters = reciters.filter(
  reciter => reciter.style === "Murattal"
);

// Get reciter by name
const mishary = reciters.find(
  reciter => reciter.reciterName.includes("Mishary")
);

if (mishary) {
  console.log(`Reciter: ${mishary.reciterName}`);
  console.log(`Style: ${mishary.style}`);
  console.log(`Translated name: ${mishary.translatedName?.name}`);
}
```

### Chapter Audio

```ts
// Get full chapter audio
const chapterAudio = await client.audio.getChapterRecitations("1", {
  reciter: 7
});

chapterAudio.forEach(recitation => {
  console.log(`Chapter ${recitation.chapterId}`);
  console.log(`File size: ${recitation.fileSize} bytes`);
  console.log(`Audio URL: ${recitation.audioUrl}`);
});
```
