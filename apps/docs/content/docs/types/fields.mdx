## Usage Examples

### Verse Field Selection

```ts
import type { VerseField } from "@quranjs/api";

// Select specific verse fields
const verse = await client.verses.findByKey("2:255", {
  fields: {
    textUthmani: true,
    textImlaei: false,
    imageUrl: true,
    codeV1: true,
    pageNumber: true
  } as Partial<Record<VerseField, boolean>>
});

// Only requested fields will be included in response
console.log(verse.textUthmani); // ✓ Available
console.log(verse.imageUrl);    // ✓ Available
console.log(verse.textImlaei);  // ✗ Undefined (not requested)
```

### Word Field Selection

```ts
import type { WordField } from "@quranjs/api";

// Get verses with specific word fields
const verse = await client.verses.findByKey("1:1", {
  words: true,
  wordFields: {
    textUthmani: true,
    textIndopak: true,
    location: true,
    audioUrl: true,
    translation: true
  } as Partial<Record<WordField, boolean>>
});

// Process word data with selected fields
verse.words?.forEach(word => {
  console.log(`Arabic: ${word.textUthmani}`);
  console.log(`Location: ${word.location}`);
  console.log(`Translation: ${word.translation?.text}`);

  if (word.audioUrl) {
    console.log(`Audio: ${word.audioUrl}`);
  }
});
```

### Translation Field Selection

```ts
import type { TranslationField } from "@quranjs/api";

// Get translations with specific fields
const verse = await client.verses.findByKey("2:255", {
  translations: [20, 131], // Multiple translations
  translationFields: {
    resourceName: true,
    languageName: true,
    verseKey: true,
    text: true
  } as Partial<Record<TranslationField, boolean>>
});

verse.translations?.forEach(translation => {
  console.log(`Resource: ${translation.resourceName}`);
  console.log(`Language: ${translation.languageName}`);
  console.log(`Verse: ${translation.verseKey}`);
  console.log(`Text: ${translation.text}`);
});
```

### Dynamic Field Selection

```ts
import type { VerseField, WordField } from "@quranjs/api";

interface FieldConfig {
  verse: Partial<Record<VerseField, boolean>>;
  word?: Partial<Record<WordField, boolean>>;
  includeAudio?: boolean;
  includeTranslations?: boolean;
}

function createFieldConfig(userPreferences: {
  scriptType: 'uthmani' | 'imlaei' | 'indopak';
  includeImages: boolean;
  includeAudio: boolean;
  showWordByWord: boolean;
}): FieldConfig {
  const config: FieldConfig = {
    verse: {
      verseNumber: true,
      verseKey: true
    }
  };

  // Add script field based on preference
  switch (userPreferences.scriptType) {
    case 'uthmani':
      config.verse.textUthmani = true;
      break;
    case 'imlaei':
      config.verse.textImlaei = true;
      break;
    case 'indopak':
      config.verse.textIndopak = true;
      break;
  }

  // Add optional fields
  if (userPreferences.includeImages) {
    config.verse.imageUrl = true;
    config.verse.imageWidth = true;
  }

  if (userPreferences.showWordByWord) {
    config.word = {
      textUthmani: true,
      translation: true,
      position: true
    };

    if (userPreferences.includeAudio) {
      config.word.audioUrl = true;
    }
  }

  return config;
}

// Usage
const fieldConfig = createFieldConfig({
  scriptType: 'uthmani',
  includeImages: true,
  includeAudio: true,
  showWordByWord: true
});

const verse = await client.verses.findByKey("1:1", {
  fields: fieldConfig.verse,
  words: !!fieldConfig.word,
  wordFields: fieldConfig.word
});
```

const FIELD_PRESETS = {

      translation: true,
      transliteration: true,
      position: true,
      audioUrl: true,
      location: true

// Helper function to apply presets
function applyFieldPreset(preset: keyof typeof FIELD_PRESETS) {
  return FIELD_PRESETS[preset];

  ...applyFieldPreset('standard'),

### Validation and Type Safety

```ts
// Validate field selections
function validateFieldSelection<T extends Record<string, boolean>>(
  fields: Partial<T>,
  allowedFields: (keyof T)[]
): { valid: boolean; errors: string[] } {
  const errors: string[] = [];

  Object.keys(fields).forEach(field => {
    if (!allowedFields.includes(field as keyof T)) {
      errors.push(`Invalid field: ${field}`);
    }
  });

  return {
    valid: errors.length === 0,
    errors
  };
}

// Runtime field validation
const ALLOWED_VERSE_FIELDS: VerseField[] = [
  'textUthmani', 'textImlaei', 'textIndopak',
  'imageUrl', 'pageNumber', 'codeV1'
];

const userFields = { textUthmani: true, invalidField: true };
const validation = validateFieldSelection(userFields, ALLOWED_VERSE_FIELDS);

if (!validation.valid) {
  console.error('Field validation errors:', validation.errors);
}
```
