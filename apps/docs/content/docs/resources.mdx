---
title: Resources API
description: Access metadata about translations, tafsirs, reciters, and other Quran resources.
---

The Resources API provides access to metadata about various Quranic resources including translations, tafsirs (commentaries), audio reciters, languages, and other reference materials.

## Methods Overview

```ts
client.resources.findAllRecitations()         // Get all available recitations
client.resources.findAllTranslations()        // Get all translations
client.resources.findAllTafsirs()            // Get all tafsirs (commentaries)
client.resources.findAllLanguages()          // Get supported languages
client.resources.findRecitationInfo(id)      // Get recitation details
client.resources.findTranslationInfo(id)     // Get translation details
client.resources.findTafsirInfo(id)         // Get tafsir details
client.resources.findAllChapterInfos()       // Get chapter info resources
client.resources.findAllChapterReciters()    // Get chapter reciters
client.resources.findAllRecitationStyles()   // Get recitation styles
client.resources.findVerseMedia()            // Get verse media info
```

## Practical Usage Examples

### Building Resource Selectors

```ts
async function getResourceSelectors() {
  const [translations, reciters, tafsirs, languages] = await Promise.all([
    client.resources.findAllTranslations(),
    client.resources.findAllChapterReciters(),
    client.resources.findAllTafsirs(),
    client.resources.findAllLanguages()
  ]);

  return {
    translationOptions: translations.map(t => ({
      id: t.id,
      name: t.name,
      language: t.languageName,
      author: t.authorName
    })),

    reciterOptions: reciters.map(r => ({
      id: r.id,
      name: r.name,
      arabicName: r.arabicName
    })),

    tafsirOptions: tafsirs.map(t => ({
      id: t.id,
      name: t.name,
      author: t.authorName,
      language: t.languageName
    })),

    languageOptions: languages.map(l => ({
      id: l.id,
      name: l.name,
      nativeName: l.nativeName,
      iso: l.iso,
      direction: l.direction
    }))
  };
}
```

### Popular Resources Helper

```ts
async function getPopularResources() {
  const [translations, reciters] = await Promise.all([
    client.resources.findAllTranslations(),
    client.resources.findAllChapterReciters()
  ]);

  // Popular English translations
  const popularEnglishTranslations = translations
    .filter(t => t.languageName === "english")
    .filter(t => ["sahih", "pickthall", "yusuf-ali"].includes(t.slug));

  // Popular Arabic reciters
  const popularReciters = reciters
    .filter(r => [1, 2, 3, 4, 5].includes(r.id)); // Top 5 reciters

  return {
    translations: popularEnglishTranslations,
    reciters: popularReciters
  };
}
```

### Resource Information Display

```ts
async function getDetailedResourceInfo(type: string, id: string) {
  switch (type) {
    case "translation":
      return await client.resources.findTranslationInfo(id);
    case "recitation":
      return await client.resources.findRecitationInfo(id);
    case "tafsir":
      return await client.resources.findTafsirInfo(id);
    default:
      throw new Error("Invalid resource type");
  }
}

// Usage
const translationDetails = await getDetailedResourceInfo("translation", "20");
const reciterDetails = await getDetailedResourceInfo("recitation", "2");
```

### Language-Specific Resources

```ts
async function getResourcesByLanguage(languageIso: string) {
  const [translations, tafsirs, languages] = await Promise.all([
    client.resources.findAllTranslations(),
    client.resources.findAllTafsirs(),
    client.resources.findAllLanguages()
  ]);

  const language = languages.find(l => l.iso === languageIso);

  return {
    language,
    translations: translations.filter(t => t.languageName === language?.name.toLowerCase()),
    tafsirs: tafsirs.filter(t => t.languageName === language?.name.toLowerCase()),
    count: {
      translations: translations.filter(t => t.languageName === language?.name.toLowerCase()).length,
      tafsirs: tafsirs.filter(t => t.languageName === language?.name.toLowerCase()).length
    }
  };
}

// Get Arabic resources
const arabicResources = await getResourcesByLanguage("ar");

// Get English resources
const englishResources = await getResourcesByLanguage("en");
```

## Error Handling

```ts
try {
  // Invalid resource ID
  const info = await client.resources.findTranslationInfo("999");
} catch (error) {
  console.error("Resource not found:", error.message);
}

// Check if resources are available
const translations = await client.resources.findAllTranslations();
if (translations.length === 0) {
  console.log("No translations available");
}
```
