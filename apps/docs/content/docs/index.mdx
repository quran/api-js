## Quickstart

```ts
import { Language, QuranClient } from "@quranjs/api";

const client = new QuranClient({
  clientId: process.env.QURAN_CLIENT_ID,
  clientSecret: process.env.QURAN_CLIENT_SECRET,
  // Optional defaults for all requests
  defaults: {
    language: Language.ENGLISH,
  },
});

// Fetch all chapters
const chapters = await client.chapters.findAll();

// Fetch a verse by key
const ayah = await client.verses.findByKey("2:255");
```

## Runtime configuration

```ts
// Read current config
const current = client.getConfig();

// Update base URL or defaults at runtime
client.updateConfig({
  contentBaseUrl: "https://apis.quran.foundation",
});

// Clear cached OAuth token (force re-auth)
client.clearCachedToken();
```
